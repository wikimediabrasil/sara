274def process_all_references(input_string): 

275 """ 

276 Gets the input_string of the done activities of a metric and dewikify all the references listed. 

277 All the references are in the format <ref name="sara-123">XYZ</ref>. 

278 """ 

279 updated_references = [] 

280 re.sub(r'<ref name="sara-\d+">.*?</ref>', lambda match: unwikify_link(match, updated_references), input_string) 

281 return updated_references

284def unwikify_link(match, updated_references): 

285 """ 

286 Receives a match for a reference and makes it into a html element. 

287 """ 

288 

289 link = match.group(0) 

290 

291 # Get the Reference ID 

292 match_ref = re.search(r'<ref name="sara-(\d+)">(.*)</ref>', link) 

293 

294 if match_ref: 

295 ref_id = match_ref.group(1) 

296 ref_content = match_ref.group(2) 

297 updated_content = replace_with_links(ref_content) # Process the ref tag inner part 

298 if "bulleted list" in updated_content: 

299 bl_match = re.match(r"(.*)\{\{bulleted list\|(.*)\}\}(.*)", updated_content) 

300 if bl_match: 

301 bullet_items = bl_match.group(2).split("|") 

302 # Make the concatenation of the bulleted list as an HTML element 

303 updated_content = bl_match.group(1) + "<ul>\n" + "\n".join( 

304 f"<li>{item}</li>" for item in bullet_items) + "\n</ul>" + bl_match.group(3) 

305 updated_link = f'<li id="sara-{ref_id}">{ref_id}. {updated_content}</li>' 

306 

307 updated_references.append(updated_link) 

308 return updated_link 

309 return link

312def replace_with_links(input_string): 

313 def replace(match): 

314 substring = match.group(0) 

315 

316 if substring.startswith('[[') and substring.endswith(']]'): 

317 content = substring[2:-2] 

318 meta = False 

319 if "|" in content: 

320 link, friendly = content.split("|", 1) 

321 if ":" not in link: 

322 meta = True 

323 elif any(link.startswith(item) for item in ["Media:", "Special:", "User:", "Project:", "File:", "MediaWiki:", "Template:", "Help:", "Category:"]): 

324 meta = True 

325 else: 

326 link = friendly = content 

327 meta = True 

328 

329 link = ur.quote(dewikify_url(link.replace(" ","_"), meta), safe=":/") 

330 return f'<a target="_blank" href="{link}">{friendly}</a>' 

331 elif substring.startswith('[') and substring.endswith(']'): 

332 content = substring[1:-1] 

333 if " " in content: 

334 link, friendly = content.split(" ", 1) 

335 else: 

336 link = friendly = content 

337 return f'<a target="_blank" href="{link}">{friendly}</a>' 

338 result = re.sub(r'(\[\[.*?\]\]|\[.*?\])', replace, input_string) 

339 return result

342def dewikify_url(link, meta=False): 

343 for pattern, prefix in INVERTED_PATTERNS.items(): 

344 match = re.match(pattern, link) 

345 if match: 

346 number_of_groups = len(match.groups()) 

347 lang = "" 

348 if number_of_groups == 2: 

349 lang = match.group(1) 

350 page = match.group(2) 

351 else: 

352 page = match.group(1) 

353 

354 page = ur.unquote(page) 

355 clean_page = page.replace("_", " ") 

356 clean_page = clean_page[:-1] if clean_page.endswith("/") else clean_page 

357 

358 return prefix.replace("$1", f"{lang}").replace("$2", f"{clean_page}") 

359 

360 # The link is a meta link, so no prefix 

361 if meta: 

362 return f"https://meta.wikimedia.org/wiki/{link}" 

363 else: # The link is not a proper Wiki link 

364 return f"{link}" if link != "-" else ""

371# 2. create the reference text

373def wikify_link(link): 

374 """ 

375 Receives a URL link and tries to wikify it, based on the patterns and correspondences. 

376 There are 4 scenarios for the return: 

377 1. The link is a simple non-Wiki URL, in which case, it returns the link as an external wikitext link, i.e. [link] 

378 2. The link is a mapped URL and from a non-language based Wikimedia project, in which case, it returns the link as an internal wikitext link, i.e. [[project:page|page]] 

379 3. The link is a mapped URL and from a language based Wikimedia project, in which case, it returns the link as an internal wikitext link, i.e. [[project:language:page|page]] 

380 4. The link is a mapped URL and from toolforge, in which case, it returns the link as an internal wikitext link, i.e. [[toolforge:project:page|page]] 

381 """ 

382 for pattern, prefix in PATTERNS.items(): 

383 match = re.match(pattern, link) 

384 if match: 

385 number_of_groups = len(match.groups()) 

386 project = "" 

387 lang = "" 

388 if number_of_groups == 3: 

389 project = match.group(1) 

390 page = match.group(3) 

391 elif number_of_groups == 2:
